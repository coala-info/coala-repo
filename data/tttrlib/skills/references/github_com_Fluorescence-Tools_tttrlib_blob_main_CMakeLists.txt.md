CMAKE\_MINIMUM\_REQUIRED(VERSION 3.13)

SET(PROJECT\_NAME tttrlib)

PROJECT(${PROJECT\_NAME})

# C++ standard

ENABLE\_LANGUAGE(CXX)

SET(CMAKE\_CXX\_STANDARD 17)

ADD\_DEFINITIONS(-D\_LIBCPP\_ENABLE\_CXX17\_REMOVED\_FEATURES)

# CMAKE

###########

SET(CMAKE\_MODULE\_PATH "${CMAKE\_SOURCE\_DIR}/cmake;${CMAKE\_MODULE\_PATH}")

MESSAGE(STATUS "CMAKE\_MODULE\_PATH='${CMAKE\_MODULE\_PATH}'")

################################################################################

# BUILD OPTIONS

################################################################################

# Default WITH\_AVX to OFF for conda/pip builds (safer, wider compatibility)

# Users can enable with: cmake -DWITH\_AVX=ON

OPTION(WITH\_AVX "Compile tttrlib with AVX intrinsics (default OFF for conda/pip builds)" OFF)

OPTION(INSTALL "Install the library and the include files" ON)

OPTION(BUILD\_PHOTON\_HDF "Build photon hdf5 reader" ON)

OPTION(BUILD\_LIBRARY "Build the library and the include files" ON)

OPTION(BUILD\_PYTHON\_INTERFACE "Build Python interface" ON)

OPTION(BUILD\_PYTHON\_DOCS "Build documentation of SWIG Python interface" OFF)

OPTION(BUILD\_R\_INTERFACE "Build R interface" OFF)

OPTION(VERBOSE\_TTTRLIB "Verbose tttrlib build" OFF)

## System-specific adjustments

################################

## System specific compiler flags

################################

# AVX - Only enable on Intel x86\_64 machines (all platforms)

# Disable on ARM (aarch64, arm64) and other non-x86\_64 architectures

IF(NOT CMAKE\_HOST\_SYSTEM\_PROCESSOR MATCHES "^(x86\_64|AMD64)$")

set(WITH\_AVX OFF CACHE STRING "Disable AVX on non-x86\_64 architectures (ARM, etc.)." FORCE)

MESSAGE(STATUS "SIMD/AVX disabled: Not an x86\_64 machine (detected: ${CMAKE\_HOST\_SYSTEM\_PROCESSOR})")

ELSE()

IF(WITH\_AVX)

MESSAGE(STATUS "SIMD/AVX enabled: (${CMAKE\_SYSTEM\_NAME})")

ELSE()

MESSAGE(STATUS "SIMD/AVX disabled via WITH\_AVX=OFF (${CMAKE\_SYSTEM\_NAME})")

ENDIF()

ENDIF()

IF(WITH\_AVX)

INCLUDE(simd)

ENDIF()

IF(WITH\_AVX)

ADD\_COMPILE\_DEFINITIONS(TTTRLIB\_WITH\_AVX=1)

ELSE()

ADD\_COMPILE\_DEFINITIONS(TTTRLIB\_WITH\_AVX=0)

ENDIF()

# OpenMP - Works on all architectures (x86\_64, ARM, etc.)

OPTION(WITH\_OPENMP "Enable OpenMP parallelization for correlation" ON)

IF(WITH\_OPENMP)

FIND\_PACKAGE(OpenMP)

IF(OpenMP\_CXX\_FOUND)

MESSAGE(STATUS "OpenMP enabled: ${OpenMP\_CXX\_VERSION} (${CMAKE\_HOST\_SYSTEM\_PROCESSOR})")

SET(CMAKE\_CXX\_FLAGS "${CMAKE\_CXX\_FLAGS} ${OpenMP\_CXX\_FLAGS}")

SET(CMAKE\_EXE\_LINKER\_FLAGS "${CMAKE\_EXE\_LINKER\_FLAGS} ${OpenMP\_EXE\_LINKER\_FLAGS}")

ELSE()

MESSAGE(WARNING "OpenMP requested but not found")

SET(WITH\_OPENMP OFF)

ENDIF()

ELSE()

MESSAGE(STATUS "OpenMP disabled")

ENDIF()

################################################################################

# HDF5 Support (Mandatory for Photon HDF)

################################################################################

IF(BUILD\_PHOTON\_HDF)

FIND\_PACKAGE(HDF5 COMPONENTS C)

ENDIF()

# Build type specific optimizations

IF (CMAKE\_BUILD\_TYPE STREQUAL "Debug")

SET(VERBOSE\_TTTRLIB ON)

MESSAGE(STATUS "Debug build: Enabling all warnings")

if(MSVC)

ADD\_COMPILE\_OPTIONS(/Wall /bigobj)

ELSE(MSVC)

ADD\_COMPILE\_OPTIONS(-Wall -Wextra -pedantic)

ENDIF(MSVC)

ELSEIF (CMAKE\_BUILD\_TYPE STREQUAL "Release")

# Aggressive optimizations for Release builds

MESSAGE(STATUS "Release build: Enabling aggressive optimizations and disabling warnings")

if(MSVC)

# Remove default warning levels to avoid D9025 warning

string(REGEX REPLACE "/W[0-4]" "" CMAKE\_CXX\_FLAGS "${CMAKE\_CXX\_FLAGS}")

string(REGEX REPLACE "/W[0-4]" "" CMAKE\_C\_FLAGS "${CMAKE\_C\_FLAGS}")

# MSVC optimizations

ADD\_COMPILE\_OPTIONS(

/O2 # Maximum optimization (speed)

/Ob2 # Inline function expansion

/Oi # Enable intrinsic functions

/Ot # Favor fast code

/GL # Whole program optimization

/w # Disable all warnings in Release builds

)

# Add AVX2 flag only if explicitly enabled by user

IF(WITH\_AVX)

ADD\_COMPILE\_OPTIONS(/arch:AVX2)

MESSAGE(STATUS "MSVC: AVX2 flag enabled via /arch:AVX2")

ELSE()

MESSAGE(STATUS "MSVC: AVX2 flag disabled (use -DWITH\_AVX=ON to enable)")

ENDIF()

# Suppress informational pragma messages from external libraries (HighFive)

ADD\_DEFINITIONS(-D\_CRT\_SECURE\_NO\_WARNINGS)

SET(CMAKE\_CXX\_FLAGS "${CMAKE\_CXX\_FLAGS} /wd4996") # Disable deprecation warnings

SET(CMAKE\_EXE\_LINKER\_FLAGS\_RELEASE "${CMAKE\_EXE\_LINKER\_FLAGS\_RELEASE} /LTCG") # Link-time code generation

SET(CMAKE\_SHARED\_LINKER\_FLAGS\_RELEASE "${CMAKE\_SHARED\_LINKER\_FLAGS\_RELEASE} /LTCG")

ELSE()

# GCC/Clang optimizations

ADD\_COMPILE\_OPTIONS(

-O3 # Maximum optimization

-funroll-loops # Unroll loops

-finline-functions # Inline functions

-fomit-frame-pointer # Omit frame pointer

-flto # Link-time optimization

-w # Disable all warnings in Release builds

)

SET(CMAKE\_EXE\_LINKER\_FLAGS\_RELEASE "${CMAKE\_EXE\_LINKER\_FLAGS\_RELEASE} -flto")

SET(CMAKE\_SHARED\_LINKER\_FLAGS\_RELEASE "${CMAKE\_SHARED\_LINKER\_FLAGS\_RELEASE} -flto")

ENDIF()

ENDIF ()

# Operation sytem specific flags

IF (MSVC)

ADD\_DEFINITIONS(-DMS\_WIN64)

ADD\_DEFINITIONS(-D\_USE\_MATH\_DEFINES)

# Issue with MSVC exceptions (https://stackoverflow.com/questions/6832666/lnk2019-when-including-asio-headers-solution-generated-with-cmake)

# https://github.com/microsoft/vcpkg/discussions/25414

ADD\_DEFINITIONS(/DWIN32 /D\_WINDOWS /GR /EHsc)

ENDIF (MSVC)

if(APPLE)

FIND\_PACKAGE(Threads)

# comatibility with older systems

set(CMAKE\_C\_FLAGS "${CMAKE\_C\_FLAGS} -mmacosx-version-min=10.13")

set(CMAKE\_CXX\_FLAGS "${CMAKE\_CXX\_FLAGS} -mmacosx-version-min=10.13")

set(CMAKE\_LFLAGS "${CMAKE\_LFLAGS} -mmacosx-version-min=10.13")

SET(CMAKE\_CXX\_FLAGS "${CMAKE\_CXX\_FLAGS} -pthread")

SET(CMAKE\_EXE\_LINKER\_FLAGS "${CMAKE\_EXE\_LINKER\_FLAGS} -pthread")

# explicit link to libc++ for recent osx versions

MESSAGE(STATUS "Explicitly link to libc++ on modern osx versions")

ADD\_DEFINITIONS(-stdlib=libc++)

SET(CMAKE\_LFLAGS "${CMAKE\_LFLAGS} -mmacosx-version-min=10.9")

# Don't set MACOSX\_R
