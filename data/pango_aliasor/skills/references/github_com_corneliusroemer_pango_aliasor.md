Usage from pango_aliasor . aliasor import Aliasor # Initalize aliasor (only needs to be done once) # If no alias_key.json is passed, downloads the latest version from github aliasor = Aliasor () # To use custom alias_key.json, pass the relative path to the file # aliasor = Aliasor('alias_key.json') # Go from aliased lineage to unaliased lineage aliasor . uncompress ( "BA.5" ) # 'B.1.1.529.5' aliasor . uncompress ( "BE.5" ) # 'B.1.1.529.5.3.1.5' aliasor . uncompress ( "XA" ) # 'XA' # Go from unaliased lineage to aliased lineage aliasor . compress ( "B.1.1.529.3.1" ) # 'BA.3.1' # Find parent lineage aliasor . parent ( "BQ.1" ) # 'BE.1.1.1' aliasor . partial_compress ( "B.1.1.529.3.1" , up_to = 1 ) # 'BA.3.1' aliasor . partial_compress ( "B.1.1.529.3.1.1.2" , up_to = 1 ) # 'BA.3.1.1.2' aliasor . partial_compress ( "B.1.1.529.3.1" , accepted_aliases = [ "AY" ]) # 'B.1.1.529.3.1' aliasor . partial_compress ( "B.1.617.2" , accepted_aliases = [ "AY" ]) # 'AY.2' aliasor . partial_compress ( 'B.1.1.529.2.75.1.2' , up_to = 4 , accepted_aliases = { "BA" }) == 'BL.2' See tests for more examples.
