# Initialize autoconf/automake environment

#######################################################

AC\_PREREQ(2.62)

AC\_REVISION([$Revision: 1.1 $])

AC\_COPYRIGHT()

AC\_INIT([mseqtools], [0.9.1], [http://www.arumugamlab.org])

# we need the aux\_dir for installing config.guess to bin/ later.

# So store it in a variable now, and use AC\_SUBST later

AC\_CONFIG\_AUX\_DIR([build-aux])

aux\_files\_dir=build-aux

#######################################################

# Define missing things: macro definitions and such

#######################################################

# AS\_VAR\_APPEND

dnl autoconf versions before 2.63b don't have AS\_VAR\_APPEND

m4\_ifdef([AS\_VAR\_APPEND],,[as\_fn\_append () { eval $[1]=\$$[1]\$[2]; }

AC\_DEFUN([AS\_VAR\_APPEND],[as\_fn\_append $1 $2])])dnl

# automake

AM\_INIT\_AUTOMAKE([-Wall gnu foreign tar-pax])

# Compilers etc

AC\_PROG\_CC

AC\_PROG\_RANLIB

AC\_LANG([C])

m4\_pattern\_allow([AM\_PROG\_AR])

AM\_PROG\_AR

#######################################################

# gzip is a mandatory requirement

#######################################################

AC\_CHECK\_PROG([GZIP], [gzip], [found], [notfound])

if test "x$GZIP" = "xnotfound"; then

AC\_MSG\_ERROR([gzip not found in path])

fi

# Need wget/curl to download kseq.h

AC\_CHECK\_PROGS([WGET], [wget curl], [notfound])

if test "x$WGET" = "xnotfound"; then

AC\_MSG\_ERROR([wget or curl not found in path])

elif test "x$WGET" = "xcurl"; then

AC\_SUBST([WGET\_ARGS], ["--location --silent -o -"])

else

AC\_SUBST([WGET\_ARGS], ["--quiet -O -"])

fi

#######################################################

# programs for installation

#######################################################

AC\_PROG\_SED

AC\_PROG\_GREP

AC\_PROG\_INSTALL

#######################################################

# C dependencies

#######################################################

# argtable

AC\_CHECK\_LIB(argtable2, arg\_parse, [], [argtable2\_devel=notfound])

if test "x$argtable2\_devel" = "xnotfound"; then

AC\_MSG\_ERROR([argtable2-devel not found])

fi

# libz

AC\_CHECK\_LIB(z, inflate, [], [zlib\_devel=notfound])

if test "x$zlib\_devel" = "xnotfound"; then

AC\_MSG\_ERROR([zlib-devel not found])

fi

# log from libm

AC\_SEARCH\_LIBS([log], [m], [],

[AC\_MSG\_ERROR([log() not found

mseqtools requires a working floating-point math library.

FAILED. This error must be resolved in order to build mseqtools successfully.])])

#######################################################

# Figure out host type

#######################################################

# We use AC\_CANONICAL\_HOST to figure it out.

# It requires that we distribute config.guess, config.sub and install.sh with the distro.

AC\_CANONICAL\_HOST

AC\_SUBST([MYCFLAGS], ["-W -Wall -Wwrite-strings -std=gnu99 -pedantic"])

AC\_SUBST([MACHINE\_TYPE], [$host\_cpu])

kernel\_name=`uname -s`

AC\_SUBST([KERNEL\_NAME], [$kernel\_name])

#######################################################

# user specified input through environmental variables

#######################################################

# substitutions

# local directories

AC\_SUBST([HOST], [generic])

AC\_SUBST([AUX\_DIR], [$aux\_files\_dir])

#######################################################

# Make outputs

#######################################################

AC\_CONFIG\_FILES([Makefile])

AC\_OUTPUT

#######################################################

# Tell the user you are done!

#######################################################

echo "

-------------------------------------------------------------------------------

mseqtools was successfully configured with the following options:

Host type : $host

Install directory : $prefix

You can build mseqtools executables by typing:

make

You can install mseqtools now by typing:

make install

This will install the executables in $prefix/bin

-------------------------------------------------------------------------------

"
