#ifndef RB3\_ALIGN\_H

#define RB3\_ALIGN\_H

#include "fm-index.h"

#define RB3\_SWF\_E2E 0x1

#define RB3\_SWF\_HAPDIV 0x2

#define RB3\_SWF\_KEEP\_RS 0x4

typedef struct {

uint32\_t flag;

int32\_t n\_best;

int32\_t min\_sc, end\_len, min\_mem\_len, max\_pos;

int32\_t match, mis;

int32\_t e2e\_drop;

int32\_t gap\_open, gap\_ext;

int32\_t r2cache\_size;

} rb3\_swopt\_t;

typedef struct {

int32\_t score;

int32\_t qlen, rlen;

int32\_t n\_cigar, cs\_len;

int32\_t n\_qoff, n\_pos, blen, mlen;

int64\_t lo, hi; // SA interval

uint8\_t \*rseq; // reference sequence in the alignment

uint32\_t \*cigar; // cigar in the BAM encoding

int32\_t \*qoff; // list of query offsets for the same hit

char \*cs;

rb3\_pos\_t \*pos;

} rb3\_swhit\_t;

typedef struct {

int32\_t n;

rb3\_swhit\_t \*a;

} rb3\_swrst\_t;

#define RB2\_SW\_MAX\_ED 6

typedef struct {

int32\_t n\_al, max\_ed, n\_hap[RB2\_SW\_MAX\_ED + 1];

} rb3\_hapdiv\_t;

void rb3\_swopt\_init(rb3\_swopt\_t \*opt);

void rb3\_sw(void \*km, const rb3\_swopt\_t \*opt, const rb3\_fmi\_t \*f, int len, const uint8\_t \*seq, rb3\_swrst\_t \*rst);

void rb3\_hapdiv(void \*km, const rb3\_swopt\_t \*opt, const rb3\_fmi\_t \*f, int len, const uint8\_t \*seq, rb3\_hapdiv\_t \*hd);

void rb3\_swrst\_free(rb3\_swrst\_t \*rst);

#endif
