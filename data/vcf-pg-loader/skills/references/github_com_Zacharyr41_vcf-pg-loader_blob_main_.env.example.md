# vcf-pg-loader Environment Variables

# Copy this file to .env and fill in your values

# NEVER commit .env with real credentials

# =============================================================================

# DATABASE CONNECTION (Option 1: Individual Components)

# =============================================================================

# Database password (REQUIRED for HIPAA compliance)

# Must be provided via environment variable, not in connection URL

VCF\_PG\_LOADER\_DB\_PASSWORD=

# Standard PostgreSQL connection variables

PGHOST=localhost

PGPORT=5432

PGUSER=postgres

PGDATABASE=variants

# =============================================================================

# DATABASE CONNECTION (Option 2: URL without password)

# =============================================================================

# Full connection URL (password will be loaded from VCF\_PG\_LOADER\_DB\_PASSWORD)

# Do NOT include password in this URL

# POSTGRES\_URL=postgresql://postgres@localhost:5432/variants

# =============================================================================

# ALTERNATIVE PASSWORD SOURCE

# =============================================================================

# If VCF\_PG\_LOADER\_DB\_PASSWORD is not set, falls back to PGPASSWORD

# PGPASSWORD=

# =============================================================================

# AWS SECRETS MANAGER (Optional)

# =============================================================================

# AWS region for Secrets Manager

# AWS\_DEFAULT\_REGION=us-east-1

# Secret name/path prefix

# VCF\_PG\_LOADER\_AWS\_SECRET\_PREFIX=vcf-pg-loader/

# =============================================================================

# HASHICORP VAULT (Optional)

# =============================================================================

# Vault server address

# VAULT\_ADDR=http://localhost:8200

# Vault authentication token

# VAULT\_TOKEN=
