use String::Diff qw( diff_fully diff diff_merge diff_regexp );# export functions # simple diff my($old, $new) = String::Diff::diff('this is Perl', 'this is Ruby'); print "$old\n";# this is [Perl] print "$new\n";# this is {Ruby} my $diff = String::Diff::diff('this is Perl', 'this is Ruby'); print "$diff-&gt;[0]\n";# this is [Perl] print "$diff-&gt;[1]\n";# this is {Ruby} my $diff = String::Diff::diff('this is Perl', 'this is Ruby', remove_open =&gt; '&lt;del&gt;', remove_close =&gt; '&lt;/del&gt;', append_open =&gt; '&lt;ins&gt;', append_close =&gt; '&lt;/ins&gt;', ); print "$diff-&gt;[0]\n";# this is &lt;del&gt;Perl&lt;/del&gt; print "$diff-&gt;[1]\n";# this is &lt;ins&gt;Ruby&lt;/ins&gt; # merged my $diff = String::Diff::diff_merge('this is Perl', 'this is Ruby'); print "$diff\n";# this is [Perl]{Ruby} my $diff = String::Diff::diff_merge('this is Perl', 'this is Ruby', remove_open =&gt; '&lt;del&gt;', remove_close =&gt; '&lt;/del&gt;', append_open =&gt; '&lt;ins&gt;', append_close =&gt; '&lt;/ins&gt;', ); print "$diff\n";# this is &lt;del&gt;Perl&lt;/del&gt;&lt;ins&gt;Ruby&lt;/ins&gt; # change to default marks %String::Diff::DEFAULT_MARKS = ( remove_open =&gt; '&lt;del&gt;', remove_close =&gt; '&lt;/del&gt;', append_open =&gt; '&lt;ins&gt;', append_close =&gt; '&lt;/ins&gt;', separator =&gt; '&amp;lt;-OLD|NEW-&amp;gt;', # for diff_merge ); # generated for regexp my $diff = String::Diff::diff_regexp('this is Perl', 'this is Ruby'); print "$diff\n";# this\ is\ (?:Perl|Ruby) # detailed list my $diff = String::Diff::diff_fully('this is Perl', 'this is Ruby'); for my $line (@{ $diff-&gt;[0] }) { print "$line-&gt;[0]: '$line-&gt;[1]'\n"; } # u: 'this is ' # -: 'Perl' for my $line (@{ $diff-&gt;[1] }) { print "$line-&gt;[0]: '$line-&gt;[1]'\n"; } # u: 'this is ' # +: 'Ruby' DESCRIPTION String::Diff is the difference of a consecutive string is made. after general diff is done, the difference in the line is searchable. the mark of the addition and the deletion can be freely changed. the color is colored to the terminal with ANSI, using the HTML display it. after the line is divided, diff is taken when 'linebreak' option is specified. my($old_string, $new_string) = String::Diff::diff_fully('this is Perl', 'this is Ruby', linebreak =&gt; 1); my($old_string, $new_string) = String::Diff::diff('this is Perl', 'this is Ruby', linebreak =&gt; 1); my $string = String::Diff::diff_merge('this is Perl', 'this is Ruby', linebreak =&gt; 1); my $string = String::Diff::diff_regexp('this is Perl', 'this is Ruby', linebreak =&gt; 1); In diff and diff_merge methods the mark of the difference can be changed. my $diff = String::Diff::diff('this is Perl', 'this is Ruby', remove_open =&gt; '&lt;del&gt;', remove_close =&gt; '&lt;/del&gt;', append_open =&gt; '&lt;ins&gt;', append_close =&gt; '&lt;/ins&gt;', ); You can escape callback set to diff function and diff_merge function. use HTML::Entities my($diff_old, $diff_new) = String::Diff::diff( 'this is &lt;b&gt;Perl&lt;/b&gt;', 'this is &lt;b&gt;&lt;BIG&gt;R&lt;/BIG&gt;uby&lt;/b&gt;', remove_open =&gt; '&lt;del&gt;', remove_close =&gt; '&lt;/del&gt;', append_open =&gt; '&lt;ins&gt;', append_close =&gt; '&lt;/ins&gt;', escape =&gt; sub { encode_entities($_[0]) }, }); is($diff_old, 'this is &amp;lt;b&amp;gt;&lt;del&gt;Perl&lt;/del&gt;&amp;lt;/b&amp;gt;'); is($diff_new, 'this is &amp;lt;b&amp;gt;&lt;ins&gt;&amp;lt;BIG&amp;gt;R&amp;lt;/BIG&amp;gt;uby&lt;/ins&gt;&amp;lt;/b&amp;gt;'); METHODS diff_fully the list that divides diff according to the mark is returnd. my($old_string, $new_string) = String::Diff::diff_fully('this is Perl', 'this is Ruby'); diff abd the mark of the deletion and the addition is given to the string. diff_merge old and new string is merged with diff. diff_regexp the regular expression to which old string and new string are matched with regexp is returned.
