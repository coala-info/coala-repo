// This is a simplified version of ksort.h

#ifndef AC\_KSORT\_H

#define AC\_KSORT\_H

#include <stdlib.h>

#include <string.h>

#include <assert.h>

#define RS\_MIN\_SIZE 64

#define RS\_MAX\_BITS 8

#define KRADIX\_SORT\_INIT(name, rstype\_t, rskey, sizeof\_key) \

typedef struct { \

rstype\_t \*b, \*e; \

} rsbucket\_##name##\_t; \

void rs\_insertsort\_##name(rstype\_t \*beg, rstype\_t \*end) \

{ \

rstype\_t \*i; \

for (i = beg + 1; i < end; ++i) \

if (rskey(\*i) < rskey(\*(i - 1))) { \

rstype\_t \*j, tmp = \*i; \

for (j = i; j > beg && rskey(tmp) < rskey(\*(j-1)); --j) \

\*j = \*(j - 1); \

\*j = tmp; \

} \

} \

void rs\_sort\_##name(rstype\_t \*beg, rstype\_t \*end, int n\_bits, int s) \

{ \

rstype\_t \*i; \

int size = 1<<n\_bits, m = size - 1; \

rsbucket\_##name##\_t \*k, b[1<<RS\_MAX\_BITS], \*be = b + size; \

assert(n\_bits <= RS\_MAX\_BITS); \

for (k = b; k != be; ++k) k->b = k->e = beg; \

for (i = beg; i != end; ++i) ++b[rskey(\*i)>>s&m].e; \

for (k = b + 1; k != be; ++k) \

k->e += (k-1)->e - beg, k->b = (k-1)->e; \

for (k = b; k != be;) { \

if (k->b != k->e) { \

rsbucket\_##name##\_t \*l; \

if ((l = b + (rskey(\*k->b)>>s&m)) != k) { \

rstype\_t tmp = \*k->b, swap; \

do { \

swap = tmp; tmp = \*l->b; \*l->b++ = swap; \

l = b + (rskey(tmp)>>s&m); \

} while (l != k); \

\*k->b++ = tmp; \

} else ++k->b; \

} else ++k; \

} \

for (b->b = beg, k = b + 1; k != be; ++k) k->b = (k-1)->e; \

if (s) { \

s = s > n\_bits? s - n\_bits : 0; \

for (k = b; k != be; ++k) \

if (k->e - k->b > RS\_MIN\_SIZE) rs\_sort\_##name(k->b, k->e, n\_bits, s); \

else if (k->e - k->b > 1) rs\_insertsort\_##name(k->b, k->e); \

} \

} \

void radix\_sort\_##name(rstype\_t \*beg, rstype\_t \*end) \

{ \

if (end - beg <= RS\_MIN\_SIZE) rs\_insertsort\_##name(beg, end); \

else rs\_sort\_##name(beg, end, RS\_MAX\_BITS, (sizeof\_key - 1) \* RS\_MAX\_BITS); \

}

#endif
