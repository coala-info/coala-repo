static int check\_table(char \*name, int lmer)

{ int kmer;

FILE \*f;

f = fopen(name,"r");

if (f == NULL)

{ fprintf(stderr,"%s: Cannot find FastK table %s\n",Prog\_Name,name);

exit (1);

}

else

{ fread(&kmer,sizeof(int),1,f);

if (lmer != 0 && kmer != lmer)

{ fprintf(stderr,"%s: Kmer (%d) of table %s != %d\n",Prog\_Name,kmer,name,lmer);

exit (1);

}

fclose(f);

return (kmer);

}

}
