archs = ["universal2"]

before-all = [

"brew install hdf5 || brew upgrade hdf5 || true",

]

before-build = "pip install delocate"

repair-wheel-command = "delocate-wheel --require-archs {delocate\_archs} -w {dest\_dir} -v {wheel}"

[tool.cibuildwheel.macos.environment]

CMAKE\_PREFIX\_PATH = "$(brew --prefix hdf5)"

CMAKE\_INSTALL\_RPATH = "@loader\_path"

SKBUILD = "1"

[tool.cibuildwheel.linux]

manylinux-x86\_64-image = "manylinux\_2\_28"

archs = ["x86\_64"]

before-all = [

"dnf install -y epel-release --nogpgcheck && dnf install -y --enablerepo=powertools --nogpgcheck cmake hdf5-devel && rm -f /usr/local/bin/swig\*",

]

repair-wheel-command = "auditwheel repair --plat manylinux\_2\_28\_x86\_64 -w {dest\_dir} {wheel}"

[tool.cibuildwheel.linux.environment]

CMAKE\_INSTALL\_RPATH = "$ORIGIN"

CMAKE\_PREFIX\_PATH = "/usr"

SKBUILD = "1"

[tool.cibuildwheel.windows]

archs = ["AMD64"]

before-build = "pip install delvewheel"

# --add-libcxx is macOS only; Windows doesn't need it

repair-wheel-command = "delvewheel repair -w {dest\_dir} {wheel}"

[tool.cibuildwheel.windows.environment]

CMAKE\_GENERATOR = "Visual Studio 17 2022"

Python3\_EXECUTABLE = "$PYTHON"

Python3\_ROOT\_DIR = "$PYTHON\_PREFIX"

CMAKE\_MSVC\_RUNTIME\_LIBRARY = "MultiThreadedDLL"

# Don't use vcpkg.cmake toolchain - it has path escaping issues on Windows

# HDF5\_ROOT and CMAKE\_PREFIX\_PATH are sufficient for CMake to find HDF5

HDF5\_ROOT = "$HDF5\_ROOT"

CMAKE\_PREFIX\_PATH = "$CMAKE\_PREFIX\_PATH"
