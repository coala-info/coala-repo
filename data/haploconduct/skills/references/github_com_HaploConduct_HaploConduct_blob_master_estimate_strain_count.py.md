def check\_overlap(seq1, seq2, pos):

mismatch\_count = 0

overlap\_len = min(len(seq1)-pos, len(seq2))

for i in xrange(overlap\_len):

base1 = seq1[pos + i]

base2 = seq2[i]

if allow\_N and 'N' in [base1, base2]:

continue

if base1 != base2:

mismatch\_count += 1

return mismatch\_count

def power\_find(n):

binary = bin(n)[:1:-1]

for x in range(len(binary)):

if int(binary[x]):

result.append(2\*\*x)

return result

clip1 = 0
