This module provides ways for representation of peptides and to obtain fragments to potentially use as loops. The following example should give you an idea of what can be done: from ost import io , seq from promod3 import loop # load an example structure prot = io . BackboneList ( frag_seq , frag_residues ) n_stem = prot . residues [ frag_pos ] c_stem = prot . residues [ frag_pos + frag_length - 1 ] # extract potential loops from fragment database based on geometry frag_db = loop . LoadFragDB () fragments = frag_db . SearchDB ( n_stem , c_stem , frag_length ) print ( &quot;Num. fragments found in FragDB: %d &quot; % len ( fragments )) # compare with reference struct_db = loop . LoadStructureDB () for i in range ( len ( fragments )): # get structure from structural database bb_list = struct_db .
