This version of Escape::Houdini has been built against the commit https://github.com/vmg/houdini/commit/3e2a78a2399bf3f58253c435278df6daf0e41740 of Houdini. FUNCTIONS escape_html( $text ) escape_html( ' &lt;body class="foo"&gt; ' ); # =&gt; '&amp;lt;body class=&amp;quot;foo&amp;quot;&amp;gt;' unescape_html( $text ) unescape_html( " &amp;lt;body&amp;gt; " ); # =&gt; '&lt;body&gt;' escape_xml( $text ) escape_xml( " &lt;foo&gt; " ); # =&gt; '&amp;lt;foo&amp;gt;' escape_uri( $text ) unescape_uri( $text ) escape_url( $text ) escape_url( " http://foo.com/meh " ); # =&gt; 'http%3A%2F%2Ffoo.com%2Fmeh' unescape_url( $text ) unescape_url( ' http%3A%2F%2Ffoo.com%2Fmeh ' ); # =&gt; "http://foo.com/meh" escape_href( $text ) escape_js( $text ) escape_js( " foo \n bar " ); # =&gt; 'foo\nbar' unescape_js( $text ) escape_js( ' foo\nbar ' ); # =&gt; "foo\nbar" EXPORTS Escape::Houdini doesn't export any function by default.
