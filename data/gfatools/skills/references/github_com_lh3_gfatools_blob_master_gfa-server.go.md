func getopt(args []string, ostr string) (int, string) {

if getopt\_place == -1 { // update scanning pointer

if optind >= len(args) || args[optind][0] != '-' {

getopt\_place = -1

return -1, ""

}

if optind < len(args) {

getopt\_place = 0

}

if getopt\_place + 1 < len(args[optind]) {

getopt\_place += 1

if args[optind][getopt\_place] == '-' { // found "--"

optind += 1

getopt\_place = -1

return -1, ""

}

}

}

optopt := args[optind][getopt\_place];

getopt\_place += 1

oli, arg := strings.IndexByte(ostr, optopt), "";

if optopt == ':' || oli < 0 {

if optopt == '-' {

return -1, ""

}

if getopt\_place < 0 {

optind += 1

}

return '?', ""

}

if oli + 1 >= len(ostr) || ostr[oli+1] != ':' {

if getopt\_place < 0 || getopt\_place >= len(args[optind]) {

optind += 1

getopt\_place = -1

}

} else {

if getopt\_place >= 0 && getopt\_place < len(args[optind]) {

arg = args[optind][getopt\_place:]

} else if optind += 1; len(args) <= optind { // no arguments

getopt\_place = -1

if len(ostr) > 0 && ostr[0] == ':' {

return '?', ""

} else {

arg = args[optind]

getopt\_place = -1

optind += 1

}
