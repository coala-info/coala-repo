taxonkit list --ids '160488,511145,889517' --indent "" \

.DEFAULT\_GOAL := show-help

# Inspired by

# <http://marmelab.com/blog/2016/02/29/auto-documented-makefile.html>

# sed script explained:

# /^##/:

# \* save line in hold space

# \* purge line

# \* Loop:

# \* append newline + line to hold space

# \* go to next line

# \* if line starts with doc comment, strip comment character off and loop

# \* remove target prerequisites

# \* append hold space (+ newline) to line

# \* replace newline plus comments by `---`

# \* print line

# Separate expressions are necessary because labels cannot be delimited by

# semicolon; see <http://stackoverflow.com/a/11799865/1968>

.PHONY: show-help

show-help:

@echo "$$(tput bold)Available rules:$$(tput sgr0)"

@echo

@sed -n -e "/^## / { \

h; \

s/.\*//; \

:doc" \

-e "H; \

n; \

s/^## //; \

t doc" \

-e "s/:.\*//; \

G; \

s/\\n## /---/; \

s/\\n/ /g; \

p; \

}" ${MAKEFILE\_LIST} \

| LC\_ALL='C' sort --ignore-case \

| awk -F '---' \

-v ncol=$$(tput cols) \

-v indent=19 \

-v col\_on="$$(tput setaf 6)" \

-v col\_off="$$(tput sgr0)" \

'{ \

printf "%s%\*s%s ", col\_on, -indent, $$1, col\_off; \

n = split($$2, words, " "); \

line\_length = ncol - indent; \

for (i = 1; i <= n; i++) { \

line\_length -= length(words[i]) + 1; \

if (line\_length <= 0) { \

line\_length = ncol - indent - length(words[i]) - 1; \

printf "\n%\*s ", -indent, " "; \

} \

printf "%s ", words[i]; \

} \

printf "\n"; \

}' \

| more $(shell test $(shell uname) = Darwin && \

echo '--no-init --raw-control-chars')
