script:

- PYTHONPATH=. pytest tests

name: test-env

- external\_pull\_requests

- tags

.deploy-python-tool:

stage: deploy

image: python:3.11-slim-bullseye

environment:

name: $ENVIRONMENT\_NAME

script:

# Start ssh-agent and set identity then run deploy script

# See https://docs.gitlab.com/ee/ci/ssh\_keys/

- apt-get update

- apt-get -y install openssh-client

- eval "$(ssh-agent)"

- echo "$EVA\_ETL\_PRIVATE\_KEY" | tr -d '\r' | ssh-add -

- ssh -o StrictHostKeyChecking=no -o UserKnownHostsFile=/dev/null ${CLUSTER\_USER}@${LOGIN\_NODE} "${SOFTWARE\_PATH}/eva-opentargets/deploy.sh ${DEPLOY\_POINT} ${ENVIRONMENT\_NAME}"

dependencies:

- test
