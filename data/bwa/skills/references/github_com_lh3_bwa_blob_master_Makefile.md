WRAP\_MALLOC=-DUSE\_MALLOC\_WRAPPERS

AR= ar

DFLAGS= -DHAVE\_PTHREAD $(WRAP\_MALLOC)

LOBJS= utils.o kthread.o kstring.o ksw.o bwt.o bntseq.o bwa.o bwamem.o bwamem\_pair.o bwamem\_extra.o malloc\_wrap.o \

QSufSort.o bwt\_gen.o rope.o rle.o is.o bwtindex.o

PROG= bwa

INCLUDES=

LIBS= -lm -lz -lpthread

SUBDIRS= .

ifeq ($(shell uname -s),Linux)

LIBS += -lrt

endif

ifeq ($(shell uname -s),GNU/kFreeBSD)

LIBS += -lrt

endif

ifneq ($(asan),)

CFLAGS+=-fsanitize=address

LIBS+=-fsanitize=address -ldl

endif

.SUFFIXES:.c .o .cc

.c.o:

$(CC) -c $(CFLAGS) $(DFLAGS) $(INCLUDES) $(CPPFLAGS) $< -o $@

all:$(PROG)

bwa:libbwa.a $(AOBJS) main.o

$(CC) $(CFLAGS) $(LDFLAGS) $(AOBJS) main.o -o $@ -L. -lbwa $(LIBS)

bwamem-lite:libbwa.a example.o

$(CC) $(CFLAGS) $(LDFLAGS) example.o -o $@ -L. -lbwa $(LIBS)

libbwa.a:$(LOBJS)

$(AR) -csru $@ $(LOBJS)

clean:

rm -f gmon.out \*.o a.out $(PROG) \*~ \*.a

depend:

( LC\_ALL=C ; export LC\_ALL; makedepend -Y -- $(CFLAGS) $(DFLAGS) $(CPPFLAGS) -- \*.c )
