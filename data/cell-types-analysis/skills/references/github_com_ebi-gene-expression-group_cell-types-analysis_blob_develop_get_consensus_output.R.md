c("-d", "--cl-dictionary"),

default = NA,

type = 'character',

),

c("-e", "--exclusions"),

action = "store",

default = NA,

help = 'Cache directory path'

action = "store",

default = 'lin',

type = 'character',

make sure to select a metric with values in the [0;1] range.'

),

make\_option(

c("-k", "--include-sem-siml"),

action = "store\_true",

default = FALSE,

type = 'logical',

help = 'Should semantic similarity be included into combined score

to those with range in the [0;1] interval only.'

c("-s", "--sort-by-agg-score"),

action = "store\_true",

default = TRUE,

type = 'logical',

),

make\_option(

c("-l", "--true-labels"),

action = "store",

type = 'character',

default = NA,

help = '(OPTIONAL) Path to the true labels tsv file in case tool

c("-n", "--return-ontology-labels"),

action = "store\_true",

default = TRUE,

type = 'logical',

help = 'Should the summary table contain cell ontology ids for the top labels? Default: TRUE'

),

make\_option(

c("-j", "--return-json"),

action = "store\_true",

default = TRUE,

type = 'logical',

help = 'Should the summary table be written as a json file (as well as tsv)? Default: TRUE'

),

make\_option(

c("-o", "--summary-table-output-path"),

action = "store",

default = NA,

type = 'character',

help = 'Path to the output table with top labels and per-cell metrics in .tsv format'

),

make\_option(

c("-r", "--raw-table-output-path"),

action = "store",

default = NA,

type = 'character',

help = 'Path to the output table with all labels in .tsv format'

)

)

"summary\_table\_output\_path",

"raw\_table\_output\_path"))
