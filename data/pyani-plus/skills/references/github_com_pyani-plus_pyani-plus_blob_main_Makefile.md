# Install OS-independent conda dependencies for development

setup\_conda-dev:

@conda install --file requirements-dev.txt --yes

# Install pyani-plus (OS-dependent, not developer version)

install\_linux:

@conda install --file requirements-thirdparty-linux.txt --yes

@pip install -U -e .

install\_macos:

@conda install --file requirements-thirdparty-macos.txt --yes

@pip install -U -e .

# Set up bioconda

setup\_bioconda:

@echo "Setting up bioconda"

@conda config --add channels bioconda

@conda config --add channels conda-forge

@conda config --set channel\_priority flexible

# Set up a conda environment for pyani-plus

setup\_conda\_env: setup\_bioconda

@echo "Creating conda environment pyani-plus\_py312"

@conda create -n pyani-plus\_py312 python=3.12 --yes

# Remove the conda environment

remove\_conda\_env:

@echo "Removing the conda environment pyani-plus\_py312"

@conda remove -n pyani-plus\_py312 --all --yes

# Set up development environment (OS-dependent)

setup\_dev\_linux: setup\_conda-dev

@conda install --file requirements-thirdparty-linux.txt --yes

@pre-commit install

@pip install -U -e .

setup\_dev\_macos: setup\_conda-dev

@conda install --file requirements-thirdparty-macos.txt --yes

@pre-commit install

@pip install -U -e .
